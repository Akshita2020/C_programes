// Print number series and stop when it finds multiple of 9
#include <stdio.h>

int main() {

    for(int i=0; i<=100;i++) {
       
        if(i%9==0) {
            
            printf(" \n find the mutilples of numbers,stopping when it finds 9 %d",i);
        }
    }

    
    
    
    return 0;
}
